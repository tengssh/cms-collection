* Project overview
This is a step-by-step tutorial for the phonon calcuations of Al using [[https://www.quantum-espresso.org/][Quantum ESPRESSO]] (QE).

import matplotlib.pyplot as plt
import numpy as np

# 1. Conversion Factor from cm^-1 to meV
cm_to_mev = 0.123984

# 2. Parse al_bands.dat by splitting blocks on blank lines
with open('al_bands.dat', 'r') as f:
    raw_content = f.read()

blocks = [b.strip() for b in raw_content.split('\n\n') if b.strip()]

# 3. Load DOS Data
dos_data = np.loadtxt('al.dos', skiprows=1)
dos_freq = dos_data[:, 0] * cm_to_mev
dos_val = dos_data[:, 1]

# 4. Setup Plots
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(10, 6), sharey=True, 
                               gridspec_kw={'width_ratios': [3, 1]})

## Left Plot: Phonon Dispersion
q_path_reference = None

for block in blocks:
    mode_data = np.fromstring(block, sep='\n').reshape(-1, 2)
    q_path = mode_data[:, 0]
    freq_mev = mode_data[:, 1] * cm_to_mev
    ax1.plot(q_path, freq_mev, color='tab:blue', lw=2)
    
    if q_path_reference is None:
        q_path_reference = q_path

idx_ticks = [0, 40, 40+20, 40+20+20, 40+20+20+40, len(q_path_reference)-1]
true_ticks = [q_path_reference[i] for i in idx_ticks]
labels = [r'$\Gamma$', 'X', 'W', 'L', r'$\Gamma$', 'X']

ax1.set_xticks(true_ticks)
ax1.set_xticklabels(labels)

for tick in true_ticks:
    ax1.axvline(tick, color='gray', linestyle='--', alpha=0.5)

ax1.set_ylabel('Frequency (meV)', fontsize=12)
ax1.set_xlim(q_path_reference.min(), q_path_reference.max())
ax1.set_ylim(0, 42)  
ax1.grid(axis='y', alpha=0.3)
ax1.set_title('Phonon Dispersion', fontsize=12)

## Right Plot: Phonon DOS
ax2.plot(dos_val, dos_freq, color='tab:orange', lw=2)
ax2.fill_betweenx(dos_freq, 0, dos_val, color='tab:orange', alpha=0.3)
ax2.set_xlabel('DOS', fontsize=12)
ax2.set_xlim(0, None)
ax2.grid(axis='both', alpha=0.3)
ax2.set_title('Phonon DOS', fontsize=12)

plt.tight_layout()
plt.savefig('al_phonon_band_dos.png', dpi=300)
EOF

import matplotlib.pyplot as plt
import numpy as np

f_dat = sys.argv[1]
f_dos = sys.argv[2]

# 1. Conversion Factor from cm^-1 to meV
cm_to_mev = 0.123984

# 2. Parse al_bands.dat by splitting blocks on blank lines
with open(f_dat, 'r') as f:
    raw_content = f.read()

blocks = [b.strip() for b in raw_content.split('\n\n') if b.strip()]

# 3. Load DOS Data

dos_freq = dos_data[:, 0] * cm_to_mev
dos_val = dos_data[:, 1]

# 4. Setup Plots
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(10, 6), sharey=True, 
                               gridspec_kw={'width_ratios': [3, 1]})

## Left Plot: Phonon Dispersion
q_path_reference = None

for block in blocks:
    mode_data = np.fromstring(block, sep='\n').reshape(-1, 2)
    q_path = mode_data[:, 0]
    freq_mev = mode_data[:, 1] * cm_to_mev
    ax1.plot(q_path, freq_mev, color='tab:blue', lw=2)
    
    if q_path_reference is None:
        q_path_reference = q_path

idx_ticks = [0, 40, 40+20, 40+20+20, 40+20+20+40, len(q_path_reference)-1]
true_ticks = [q_path_reference[i] for i in idx_ticks]
labels = [r'$\Gamma$', 'X', 'W', 'L', r'$\Gamma$', 'X']

ax1.set_xticks(true_ticks)
ax1.set_xticklabels(labels)

for tick in true_ticks:
    ax1.axvline(tick, color='gray', linestyle='--', alpha=0.5)

ax1.set_ylabel('Frequency (meV)', fontsize=12)
ax1.set_xlim(q_path_reference.min(), q_path_reference.max())
ax1.set_ylim(0, 42)  
ax1.grid(axis='y', alpha=0.3)
ax1.set_title('Phonon Dispersion', fontsize=12)

## Right Plot: Phonon DOS
ax2.plot(dos_val, dos_freq, color='tab:orange', lw=2)
ax2.fill_betweenx(dos_freq, 0, dos_val, color='tab:orange', alpha=0.3)
ax2.set_xlabel('DOS', fontsize=12)
ax2.set_xlim(0, None)
ax2.grid(axis='both', alpha=0.3)
ax2.set_title('Phonon DOS', fontsize=12)

plt.tight_layout()
plt.savefig('phonon_bands_dos.png', dpi=300)
