import warnings
warnings.filterwarnings("ignore", category=UserWarning)
import ase.io
from ase import Atoms
from ase.data import atomic_masses, atomic_numbers

# 1. Extract 'a' from CIF
cif_atoms = ase.io.read('1010093.cif')
cif_a = cif_atoms.cell.cellpar()[0]

# 2. Build the correct AFM-II Rhombohedral Supercell
cell = [[ 0.5 * cif_a,  0.0 * cif_a,  0.5 * cif_a],
        [ 0.5 * cif_a,  0.5 * cif_a,  0.0 * cif_a],
        [ 0.0 * cif_a,  0.5 * cif_a,  0.5 * cif_a]]

# Correct fractional positions for this specific 4-atom magnetic cell
positions = [[0.0, 0.0, 0.0],    # Ni1 (Spin Up)
             [0.5, 0.5, 0.5],    # Ni2 (Spin Down)
             [0.25, 0.25, 0.25], # O
             [0.75, 0.75, 0.75]] # O

# Use your clever placeholders so ASE separates the types
magnetic_cell = Atoms(symbols=['He', 'Ne', 'O', 'O'], 
                      scaled_positions=positions, 
                      cell=cell, 
                      pbc=True)

# 3. Write out Quantum ESPRESSO template
pseudos = {
    'He': 'Ni.pbesol-spn-kjpaw_psl.1.0.0.UPF',
    'Ne': 'Ni.pbesol-spn-kjpaw_psl.1.0.0.UPF',
    'O':  'O.pbesol-n-kjpaw_psl.1.0.0.UPF'
}
ase.io.write('01-nio.in', magnetic_cell, format='espresso-in', 
             pseudopotentials=pseudos, crystal_coordinates=True)

# 4. Pull authentic masses dynamically
ni_mass = atomic_masses[atomic_numbers['Ni']]
o_mass = atomic_masses[atomic_numbers['O']]

# 5. Clean up strings and element tokens in the output file
with open('01-nio.in', 'r') as f:
    text = f.read()

# Replace the placeholder masses with standard precise masses
text = text.replace('He 4.002602', f'Ni1 {ni_mass:.4f}')
# Depending on ASE versions, check if it prints 4 or more decimals for Ne
text = text.replace('Ne 20.1797', f'Ni2 {ni_mass:.4f}') 
text = text.replace('O 15.999', f'O {o_mass:.4f}')

# Fix position tokens
text = text.replace('He', 'Ni1')
text = text.replace('Ne', 'Ni2')

with open('01-nio.in', 'w') as f:
    f.write(text)
EOF

import argparse
import numpy as np
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt
import subprocess

# 1. Parse command-line arguments
parser = argparse.ArgumentParser(description="Plot PDOS from Quantum ESPRESSO outputs.")
parser.add_argument("-i", "--input_log", type=str, default="nscf.out",
                    help="Path to the NSCF output log file (default: nscf.out)")
parser.add_argument("-p", "--prefix", type=str, default="prefix",
                    help="The prefix used in your projwfc.x / files (default: prefix)")
parser.add_argument("-o", "--output_fig", type=str, default="pdos.png",
                    help="Path/filename for the output figure (default: pdos.png)")
args = parser.parse_args()

# 2. Dynamically grab the true Fermi/Highest Occupied Energy from the log
fermi_energy = 0.0
try:
    # First, try parsing as an insulator (fixed occupations)
    res = subprocess.check_output(f"grep 'highest occupied, lowest unoccupied' {args.input_log}", shell=True).decode()
    # Extract the highest occupied level (the first number after the colon)
    levels = res.split(':')[1].split()
    fermi_energy = float(levels[0])
    print(f"Parsed Insulating Valence Band Max (Highest Occupied): {fermi_energy} eV")
except Exception:
    try:
        # Fallback to metallic smearing string if it's a smeared run
        res = subprocess.check_output(f"grep 'the Fermi energy is' {args.input_log}", shell=True).decode()
        fermi_energy = float(res.split('is')[1].split('ev')[0].strip())
        print(f"Parsed Metallic Fermi Energy: {fermi_energy} eV")
    except Exception:
        print(f"Could not parse energy reference from {args.input_log}, defaulting to 0.0 eV")

print(f"Using Reference Energy: {fermi_energy} eV")

# 3. Extract base energy grid from total DOS file using the dynamic prefix
tot_file = f"{args.prefix}.pdos.pdos_tot"
print(f"Reading total DOS from: {tot_file}")
tot_data = np.loadtxt(tot_file)
energy = tot_data[:, 0] - fermi_energy

# 4. Explicitly parse the specific orbital files using the dynamic prefix
try:
    ni1_file = f"{args.prefix}.pdos.pdos_atm#1(Ni1)_wfc#4(d)"
    ni2_file = f"{args.prefix}.pdos.pdos_atm#2(Ni2)_wfc#4(d)"
    
    ni1_data = np.loadtxt(ni1_file)
    ni2_data = np.loadtxt(ni2_file)
    
    ni_majority = ni1_data[:, 1] + ni2_data[:, 2]
    ni_minority = ni1_data[:, 2] + ni2_data[:, 1]

import argparse
import numpy as np
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt
import subprocess

# 1. Parse command-line arguments
parser = argparse.ArgumentParser(description="Plot PDOS from Quantum ESPRESSO outputs.")
parser.add_argument("-i", "--input_log", type=str, default="nscf.out",
                    help="Path to the NSCF output log file (default: nscf.out)")
parser.add_argument("-p", "--prefix", type=str, default="prefix",
                    help="The prefix used in your projwfc.x / files (default: prefix)")
parser.add_argument("-o", "--output_fig", type=str, default="pdos.png",
                    help="Path/filename for the output figure (default: pdos.png)")
args = parser.parse_args()

# 2. Dynamically grab the true Fermi/Highest Occupied Energy from the log
fermi_energy = 0.0
try:
    # First, try parsing as an insulator (fixed occupations)
    res = subprocess.check_output(f"grep 'highest occupied, lowest unoccupied' {args.input_log}", shell=True).decode()
    # Extract the highest occupied level (the first number after the colon)
    levels = res.split(':')[1].split()
    fermi_energy = float(levels[0])
    print(f"Parsed Insulating Valence Band Max (Highest Occupied): {fermi_energy} eV")
except Exception:
    try:
        # Fallback to metallic smearing string if it's a smeared run
        res = subprocess.check_output(f"grep 'the Fermi energy is' {args.input_log}", shell=True).decode()
        fermi_energy = float(res.split('is')[1].split('ev')[0].strip())
        print(f"Parsed Metallic Fermi Energy: {fermi_energy} eV")
    except Exception:
        print(f"Could not parse energy reference from {args.input_log}, defaulting to 0.0 eV")

print(f"Using Reference Energy: {fermi_energy} eV")

# 3. Extract base energy grid from total DOS file using the dynamic prefix
tot_file = f"{args.prefix}.pdos_tot"
print(f"Reading total DOS from: {tot_file}")
tot_data = np.loadtxt(tot_file)
energy = tot_data[:, 0] - fermi_energy

# 4. Explicitly parse the specific orbital files using the dynamic prefix
try:
    ni1_file = f"{args.prefix}.pdos_atm#1(Ni1)_wfc#4(d)"
    ni2_file = f"{args.prefix}.pdos_atm#2(Ni2)_wfc#4(d)"
    
    ni1_data = np.loadtxt(ni1_file)
    ni2_data = np.loadtxt(ni2_file)
    
    ni_majority = ni1_data[:, 1] + ni2_data[:, 2]
    ni_minority = ni1_data[:, 2] + ni2_data[:, 1]
except Exception as e:
    print(f"Failed to load Ni files with prefix {args.prefix}: {e}")
    raise e

# Oxygen p-orbitals
try:
    o3_file = f"{args.prefix}.pdos_atm#3(O)_wfc#2(p)"
    o4_file = f"{args.prefix}.pdos_atm#4(O)_wfc#2(p)"
    
    o3_data = np.loadtxt(o3_file)
    o4_data = np.loadtxt(o4_file)
    o_total = (o3_data[:, 1] + o3_data[:, 2]) + (o4_data[:, 1] + o4_data[:, 2])
except Exception as e:
    print(f"Failed to load Oxygen files with prefix {args.prefix}: {e}")
    raise e

# 5. Generate the Plot
plt.figure(figsize=(9, 6))

plt.plot(energy, ni_majority, color='red', label='Ni-3d (majority spin)', linewidth=1.2)
plt.plot(energy, ni_minority, color='limegreen', label='Ni-3d (minority spin)', linewidth=1.2)
plt.plot(energy, o_total, color='blue', label='O-2p', linewidth=1.2)

# Styling elements
plt.axvline(x=0.0, color='black', linestyle='-', linewidth=0.8)
plt.xlabel('E - E$_F$ (eV)', fontsize=12)
plt.ylabel('PDOS (states/eV/atom)', fontsize=12)
plt.xlim(-10.0, 8.0)
plt.ylim(0, None)
plt.legend(frameon=False, loc='upper right', fontsize=11)
plt.tight_layout()

# Save image file
plt.savefig(args.output_fig, dpi=300)
print(f"Figure saved successfully to: {args.output_fig}")
